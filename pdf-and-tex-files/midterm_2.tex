\textbf{Notation:}
\begin{itemize}
    \item If $n \in \mathbb{Z}_{>0}$ and $a,b \in \ZZ$ then $a \equiv b \pmod{n}$ means that $n | (b-a)$.
	\item If $n \in \mathbb{Z}_{>0}$ then $\mathbb{Z}_n = \{0,1,\ldots, n-1\}$ is the set of integers modulo $n$. 
    \item If $n \in \mathbb{Z}_{>0}$ and $a\in \ZZ$ we write $a\%n$ for the remainder of the Euclidean division of $a$ by $n$. This is unique element of $\mathbb{Z}_n$ such that $a \equiv a\% n \pmod{n}$. 
    \item If $n \in \mathbb{Z}_{>0}$ and $\gcd(a,n) = 1$, then $\text{ord}_n(a)$ is the smallest positive integer $k$ such that $a^k \equiv 1 \pmod{n}$.
\end{itemize}

\question[10] % 1: Exponentiation
\begin{parts}
    \part Find $5^{9603} \pmod{360}$.
    \vfill
    \part How many multiplications (including squarings) need to be performed to find $a^{265} \pmod{n}$ according to the binary exponentiation (square-and-multiply) algorithm?
    \vfill
\end{parts}

\newpage
\question[10] % 2: Primitive Roots and Order
\begin{parts}
    \part Find all primitive roots modulo $11$.
    \vfill
    \part If $\text{ord}_n(a) = 60$, what is $\text{ord}_n(a^{42})$?
    \vfill
\end{parts}

\newpage
\question[5] % 3: CRT
Find all solutions $x$ to the following system of congruences:
\begin{align*}
    x &\equiv 2 \pmod{9} \\
    x &\equiv 4 \pmod{7}
\end{align*}
If no solution exists, explain why.

\vfill

\newpage
\question[5] % 4: Miller-Rabin
Perform one iteration of the Miller-Rabin primality test on $n = 25$ using the base $a = 7$. State whether $25$ is a strong probably prime or if $7$ is a witness for compositeness.

\vfill

\newpage
\question[10] % 5: RSA
An RSA public key is $(n, e) = (55, 3)$. 
\begin{parts}
    \part Find the private exponent $d$.
    \vfill
    \part Decrypt the ciphertext $c = 4$.
    \vfill
\end{parts}

\newpage
\question[10] % 6: Proof
Let $n > 1$ and $\gcd(a, n) = 1$. Suppose $d = \text{ord}_n(a)$. Prove that if $a^k \equiv 1 \pmod{n}$, then $d \mid k$. 

\vfill
